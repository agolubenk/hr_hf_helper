# Автоматические backup'ы HRHelper
# Добавьте эти строки в crontab с помощью команды: crontab -e

# Ежедневный backup в 02:00
0 2 * * * cd /path/to/hrhelper/fullstack/backend && ./backup.sh create >> logs/backup.log 2>&1

# Еженедельный полный backup в воскресенье в 01:00
0 1 * * 0 cd /path/to/hrhelper/fullstack/backend && ./backup.sh create >> logs/backup_weekly.log 2>&1

# Очистка старых backup'ов каждую неделю в понедельник в 03:00
0 3 * * 1 cd /path/to/hrhelper/fullstack/backend && ./backup.sh cleanup >> logs/backup_cleanup.log 2>&1

# Мониторинг системы каждые 5 минут в рабочее время (09:00-18:00, пн-пт)
*/5 9-18 * * 1-5 cd /path/to/hrhelper/fullstack/backend && ./monitor.sh quick >> logs/monitor.log 2>&1

# Проверка здоровья системы каждый час
0 * * * * cd /path/to/hrhelper/fullstack/backend && ./monitor.sh health >> logs/health_check.log 2>&1
